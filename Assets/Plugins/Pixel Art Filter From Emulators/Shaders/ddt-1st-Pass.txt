	struct out_vertex
	{
		float4 position : POSITION;
		float2 texCoord : TEXCOORD0;
		float4 t1       : TEXCOORD1;
		float2 loc      : TEXCOORD2;

	};
	
	//VERTEX_SHADER
	out_vertex main_vertex(appdata_base v)
	{
		out_vertex OUT;

		OUT.position = UnityObjectToClipPos(v.vertex);

		float2 ps = float2(1.0 / texture_size.x, 1.0 / texture_size.y);
		float dx = ps.x;
		float dy = ps.y;




		// This line fix a bug in ATI cards.
		float2 texCoord1 = v.texcoord + float2(0.0000001, 0.0000001);

		OUT.texCoord = texCoord1;
		OUT.t1.xy = float2(dx, 0); // F
		OUT.t1.zw = float2(0, dy); // H
		OUT.loc = texCoord1*(texture_size);

		return OUT;
	}
